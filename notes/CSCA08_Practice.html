Vectors are represented as List[float], with the following properties:

  • The sum of 2 vectors both of lengths nnn, [x1, x2, ..., xn] and [y1, y2, ..., yn] is given by [x1 + y1, x2 + y2, ..., xn + yn]. Vectors of different lengths cannot be added. Note that the sum is also a List[float].
  • The dot product of 2 vectors both of lengths nnn, [x1, x2, ..., xn] and [y1, y2, ..., yn] is given by x1*y1 + x2*y2 + ... + xn*yn. This operation cannot be applied to vectors of different lengths. Note that the dot product is a float.
def add_vector(vector_1: List[float], vector_2: List[float]):
	'''
	Returns the sum of vector_1 and vector_2.
	Precondition: len(vector_1) == len(vector_2)
	>>> add_vector([1.0, 2.0], [2.0, 3.0])
	[3.0, 5.0]
	>>> add_vector([0.0, 0.0], [2.0, 3.0])
	[2.0, 3.0]
	'''
	result = []
	for i in range(len(vector_1)):
		result.append(vector_1[i] + vector_2[i])
		
	return result
	
def dot_product(vector_1: List[float], vector_2: List[float]):
	'''
	Returns the dot of this vector added to other_vector.
	Precondition: len(vector_1) == len(vector_2)
	>>> dot_product([1.0, 2.0], [2.0, 3.0])
	8.0
	>>> dot_product([0.0, 0.0], [2.0, 3.0])
	0.0
	'''
	result = 0.0
	for i in range(len(vector_1)):
		result += vector_1[i] * vector_2[i]
	
	return result

def second_largest(lst: list[int]) -> int
	'''
	Return the second largest integer in lst.
	Precondition: len(lst) >= 2
	>>> second_largest(1, 2, 3)
	2
	>>> second_largest(6,6,5)
	5
	'''
	largest = 0
	second_largest = 0
	for number in lst:
		if(number > largest):
			second_largest = largest
			largest = number
		elif(number > second_largest and number != largest):
			second_largest = number
			
	return second_largest

def line_of_best_fit(coordinates: List[Tuple]) -> Tuple:
    '''
    Return the slope and y-intercept of the line of best fit to coordinates.
    >>> line_of_best_fit([(1,1), (2,2), (3,3)])
    (1.0, 0.0)
    >>> line_of_best_fit([(1, 1), (2, 2), (3, 1.3), (4, 3.75), (5, 2.25)])
    (0.425, 0.785)
    '''

    x_bar = 0
    y_bar = 0
    for pair in coordinates:
        x_bar += pair[0]
        y_bar += pair[1]
    x_bar /= len(coordinates)
    y_bar /= len(coordinates)

    a_top = 0
    a_bot = 0
    for pair in coordinates:
        a_top += (pair[0] - x_bar) * (pair[1] - y_bar)
        a_bot += (pair[0] - x_bar) ** 2
        
    a = a_top / a_bot
    return (a, y_bar - a * x_bar)
